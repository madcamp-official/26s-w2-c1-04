# Memory Pager — GPU 서버, Stable Diffusion 전용 venv
#
#   uv venv --python 3.11 ~/envs/sd
#   source ~/envs/sd/bin/activate && pip install -r gpu/requirements-sd.txt
#
# vLLM 환경(gpu/requirements-vllm.txt)과 torch 핀이 충돌하므로 venv 를 반드시 나눈다.
# 호스트에 CUDA 툴킷을 깔지 마라. torch 가 CUDA 런타임을 wheel 에 번들한다.

torch==2.9.0        # requires Python>=3.10
diffusers==0.39.0   # 2026-07-03 최신. SD 1.5 파이프라인

# --- 아래는 버전 미확정. pip 해석에 맡기고 동작 확인 후 pip freeze 로 고정할 것. ---
transformers        # diffusers 의 텍스트 인코더 의존
accelerate          # 학습/추론 런처
peft                # LoRA 학습·로딩
bitsandbytes        # 8-bit Adam. LoRA 학습에서 VRAM 을 아낀다

# VRAM (docs/STACK.md 2절, 검증됨):
#   SD 1.5 추론 (512px, fp16, 1장)              4~6GB   -> 상주 LLM 켠 채로 가능
#   SD 1.5 LoRA 학습 (512px, batch1, 8bit Adam) 6~8GB   -> 상주 LLM 켠 채로 가능 (헤드룸 4~6GB)
#
# SDXL 은 쓰지 않는다. 단독 학습은 20GB 에 들어가지만(피크 13~15GB) 상주 LLM 과 공존이 불가능하다.
